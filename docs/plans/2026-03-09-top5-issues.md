/** Get full session state for restoration (session + sets + notes) */
export async function getFullSessionState(sessionId: string): Promise<{
  session: Session;
  setLogs: SetLog[];
  exerciseNotes: Record<string, string>;
} | null> {
  const db = await getDatabase();
  const session = await db.getFirstAsync<Session>(
    "SELECT * FROM sessions WHERE id = ?",
    [sessionId]
  );
  if (!session) return null;

  const setLogs = await db.getAllAsync<SetLog>(
    "SELECT * FROM set_logs WHERE session_id = ? ORDER BY exercise_id, set_number",
    [sessionId]
  );

  const noteRows = await db.getAllAsync<{ exercise_id: string; note: string }>(
    "SELECT exercise_id, note FROM exercise_notes WHERE session_id = ?",
    [sessionId]
  );
  const exerciseNotes: Record<string, string> = {};
  for (const row of noteRows) {
    exerciseNotes[row.exercise_id] = row.note;
  }

  return { session, setLogs, exerciseNotes };
}
```

// Inside useWorkoutSession, after loadData definition:

const restoreSession = useCallback(async (programId: string, def: any) => {
  const inProgress = await getInProgressSession(programId);
  if (!inProgress) return false;

  // Check for stale sessions (>4 hours)
  const startedMs = new Date(inProgress.started_at).getTime();
  const hoursElapsed = (Date.now() - startedMs) / (1000 * 60 * 60);

  const state = await getFullSessionState(inProgress.id);
  if (!state) return false;

  // Restore session identity
  setSessionId(inProgress.id);
  setStartedAt(inProgress.started_at);
  setSelectedDay(inProgress.day_template_id);

  // Restore readiness
  setSleep(state.session.sleep ?? 3);
  setSoreness(state.session.soreness ?? 3);
  setEnergy(state.session.energy ?? 3);

  // Restore warmup
  setWarmupRope(!!state.session.warmup_rope);
  setWarmupAnkle(!!state.session.warmup_ankle);
  setWarmupHipIr(!!state.session.warmup_hip_ir);

  // Restore exercise notes
  setExerciseNotes(state.exerciseNotes);

  // Restore session notes
  if (state.session.notes) setSessionNotes(state.session.notes);

  // Rebuild exercises from template + logged sets
  const template = def.weekly_template[inProgress.day_template_id];
  if (!template) return false;

  const weekNumber = inProgress.week_number;
  const oneRm: Record<string, number> = /* parse from program */;

  // Group logged sets by exercise_id
  const logsByExercise: Record<string, SetLog[]> = {};
  for (const log of state.setLogs) {
    if (!logsByExercise[log.exercise_id]) logsByExercise[log.exercise_id] = [];
    logsByExercise[log.exercise_id].push(log);
  }

  // Get exercise names
  const exerciseIds = [...new Set(state.setLogs.map(s => s.exercise_id))];
  const exerciseNames = await getExerciseNames(exerciseIds);

  const exStates: ExerciseState[] = [];
  for (const slot of template.exercises) {
    const target = getTargetForWeek(slot, weekNumber);
    if (!target) continue;

    const exerciseDef = def.exercise_definitions.find((e: any) => e.id === slot.exercise_id);
    const name = exerciseDef?.name ?? exerciseNames[slot.exercise_id] ?? slot.exercise_id.replace(/_/g, ' ');
    const reps = typeof target.reps === 'string' ? parseInt(target.reps) || 8 : target.reps;

    let suggestedWeight = 0;
    if (target.percent && oneRm[slot.exercise_id]) {
      const pct = typeof target.percent === 'string' ? parseFloat(target.percent) : target.percent;
      suggestedWeight = calculateTargetWeight(oneRm[slot.exercise_id], pct);
    }
    const weight = suggestedWeight || 0;

    const logged = logsByExercise[slot.exercise_id] || [];
    const sets: SetState[] = Array.from({ length: target.sets }, (_, i) => {
      const log = logged.find(l => l.set_number === i + 1);
      if (log) {
        return {
          setNumber: i + 1,
          targetWeight: log.target_weight,
          targetReps: log.target_reps,
          actualWeight: log.actual_weight ?? weight,
          actualReps: log.actual_reps ?? reps,
          rpe: log.rpe ?? undefined,
          status: log.status as SetState['status'],
          id: log.id,
        };
      }
      return {
        setNumber: i + 1,
        targetWeight: weight,
        targetReps: reps,
        actualWeight: weight,
        actualReps: reps,
        status: 'pending' as const,
      };
    });

    // Restore RPE from any completed set
    const rpeLog = logged.find(l => l.rpe != null);

    exStates.push({
      slot,
      exerciseName: name,
      sets,
      rpe: rpeLog?.rpe ?? undefined,
      expanded: false,
    });
  }

  // Also restore ad-hoc exercises (sets not matching template exercises)
  const templateExerciseIds = new Set(template.exercises.map((e: any) => e.exercise_id));
  for (const [exerciseId, logs] of Object.entries(logsByExercise)) {
    if (templateExerciseIds.has(exerciseId)) continue;
    const name = exerciseNames[exerciseId] ?? exerciseId.replace(/_/g, ' ');
    const sets: SetState[] = logs.map(log => ({
      setNumber: log.set_number,
      targetWeight: log.target_weight,
      targetReps: log.target_reps,
      actualWeight: log.actual_weight ?? 0,
      actualReps: log.actual_reps ?? 0,
      status: log.status as SetState['status'],
      id: log.id,
    }));
    exStates.push({
      slot: { exercise_id: exerciseId, category: 'accessory', targets: [] },
      exerciseName: name,
      sets,
      expanded: false,
      isAdhoc: true,
    });
  }

  // Expand first incomplete exercise
  const firstIncomplete = exStates.findIndex(e => e.sets.some(s => s.status === 'pending'));
  if (firstIncomplete >= 0) exStates[firstIncomplete].expanded = true;

  setExercises(exStates);

  // Determine phase
  const hasLoggedSets = state.setLogs.length > 0;
  setPhase(hasLoggedSets ? 'logging' : 'warmup');

  // If stale, prompt user
  if (hoursElapsed > 4) {
    Alert.alert(
      'Resume Workout?',
      `You have an unfinished workout from ${Math.round(hoursElapsed)} hours ago.`,
      [
        { text: 'Resume', style: 'default' },
        {
          text: 'Complete As-Is',
          onPress: async () => {
            await completeSession(inProgress.id, !!state.session.conditioning_done);
            setPhase('complete');
          },
        },
        {
          text: 'Discard',
          style: 'destructive',
          onPress: async () => {
            await deleteSession(inProgress.id);
            setSessionId(null);
            setPhase('select');
            setExercises([]);
            setStartedAt(null);
          },
        },
      ],
    );
  }

  return true;
}, []);

/** Stop/end an active program */
export async function stopProgram(
  programId: string,
  deleteData: boolean
): Promise<void> {
  const db = await getDatabase();

  if (deleteData) {
    // Cascade delete in FK dependency order
    await db.runAsync(
      `DELETE FROM personal_records WHERE session_id IN
       (SELECT id FROM sessions WHERE program_id = ?)`,
      [programId]
    );
    await db.runAsync(
      `DELETE FROM exercise_notes WHERE session_id IN
       (SELECT id FROM sessions WHERE program_id = ?)`,
      [programId]
    );
    await db.runAsync(
      `DELETE FROM set_logs WHERE session_id IN
       (SELECT id FROM sessions WHERE program_id = ?)`,
      [programId]
    );
    await db.runAsync(
      `DELETE FROM run_logs WHERE session_id IN
       (SELECT id FROM sessions WHERE program_id = ?)`,
      [programId]
    );
    await db.runAsync(
      'DELETE FROM sessions WHERE program_id = ?',
      [programId]
    );
    await db.runAsync(
      'DELETE FROM weekly_checkins WHERE program_id = ?',
      [programId]
    );
    // Mark as inactive (available for reactivation)
    await db.runAsync(
      "UPDATE programs SET status = 'inactive', activated_date = NULL WHERE id = ?",
      [programId]
    );
  } else {
    // Soft stop — keep all data, mark as completed
    await db.runAsync(
      "UPDATE programs SET status = 'completed' WHERE id = ?",
      [programId]
    );
  }
}

/** Apply override weight/reps to all pending sets of the exercise */
const saveOverrideToAll = async () => {
  if (!overrideModal || !sessionId) return;
  const { exerciseIdx } = overrideModal;
  const ex = exercises[exerciseIdx];

  for (let i = 0; i < ex.sets.length; i++) {
    const set = ex.sets[i];
    const hitTarget = overrideWeight >= set.targetWeight && overrideReps >= set.targetReps;
    const status: SetLog['status'] = hitTarget ? 'completed' : 'completed_below';

    if (set.id) {
      await updateSet(set.id, {
        actualWeight: overrideWeight,
        actualReps: overrideReps,
        status,
      });
    } else {
      const setId = await logSet({
        sessionId,
        exerciseId: ex.slot.exercise_id,
        setNumber: set.setNumber,
        targetWeight: set.targetWeight,
        targetReps: set.targetReps,
        actualWeight: overrideWeight,
        actualReps: overrideReps,
        status,
        isAdhoc: ex.isAdhoc,
      });
      ex.sets[i] = { ...ex.sets[i], id: setId };
    }
  }

  setExercises(prev => {
    const next = [...prev];
    next[exerciseIdx] = {
      ...next[exerciseIdx],
      sets: next[exerciseIdx].sets.map(s => ({
        ...s,
        actualWeight: overrideWeight,
        actualReps: overrideReps,
        status: (overrideWeight >= s.targetWeight && overrideReps >= s.targetReps)
          ? 'completed' as const : 'completed_below' as const,
      })),
    };
    return next;
  });

  setOverrideModal(null);
  await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
};
```

Return `saveOverrideToAll` from the hook.
